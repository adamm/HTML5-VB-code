Update the Application Cache

Synopsis
if ('applicationCache' in window) {
  var appCache = applicationCache;
  appCache.addEventListener('updateready', function(event){
    appCache.swapCache();
    if ( ! appCache.autoUpdate || confirm( "Update found. Reload?" ) )
      location.reload();
  }, false);

  setInterval(function() {
    appCache.autoUpdate = 1;
    appCache.update();
  }, 10000 );
}
