Offline Detection

Browser Status Detection

Synopsis
function browserOnline()  {
  // Browser is online
}
function browserOffline() {
  // Browser is offline
}

window.addEventListener('online', browserOnline, false);
window.addEventListener('offline', browserOffline, false);

if ( "onLine" in navigator ) {
  if ( navigator.onLine ) {
    browserOnline();
  }
  else {
    browserOffline();
  }
}
